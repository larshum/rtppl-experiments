type RoomMap = {squares : [[Bool]], squareSide : Float}
sensor frontLeft : Float
sensor frontRight : Float
// variable periodicity, varying #particles
def positioning(roomMap : RoomMap, s : Sensor(Float), period : Float) {
  sdelay(period)
}

def periodic(f : Unit -> Unit, period : Float) {
  loop {
    sdelay(period)
    f()
  }
}
